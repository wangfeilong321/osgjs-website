    var initChannel = function ( type, keys, times, targetName, channelName, channel ) {
    var start = times[ 0 ];
    var end = times[ times.length - 1 ];
    channel.start = start;
    channel.keys = keys;
    channel.end = end;
    channel.times = times;
    channel.type = type;
    channel.duration = end - start;
    channel.target = targetName;
    channel.name = channelName;
    return channel;
};
